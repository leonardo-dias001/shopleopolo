CREATE TABLE itens (
    CODIGO INT PRIMARY KEY AUTO_INCREMENT,
    TITULO VARCHAR(255),
    DESCRICAO TEXT,
    FOTO VARCHAR(255),
    VALOR_COMPRA DECIMAL(10, 2),
    VALOR_VENDA DECIMAL(10, 2),
    LUCRO DECIMAL(10, 2)
);
-------------------------------------------------------------------------------
-- Criação da tabela CARGOS
CREATE TABLE CARGOS (
  ID_CARGO INT PRIMARY KEY,
  DSC_CARGO VARCHAR(255),
  NIVEL_ACESSO INT,
  MODO_PAGAMENTO VARCHAR(255),
  VALOR_HORA DECIMAL(10,2),
  VALOR_SEMANA DECIMAL(10,2),
  VALOR_MES DECIMAL(10,2),
  VALOR_ENTREGA DECIMAL(10,2)
);

-- Criação da tabela PESSOA
CREATE TABLE PESSOA (
  ID_PESSOA INT PRIMARY KEY,
  NOME VARCHAR(255),
  CPF VARCHAR(20),
  RG VARCHAR(20),
  ENDERECO VARCHAR(255),
  NUMERO INT,
  BAIRRO VARCHAR(255),
  DATA_NASCIMENTO DATE,
  ID_CARGO INT,
  FOREIGN KEY (ID_CARGO) REFERENCES CARGOS(ID_CARGO)
);

-- Criação da tabela USUARIOS
CREATE TABLE USUARIOS (
  MATRICULA INT PRIMARY KEY,
  ID_PESSOA INT,
  NIVEL_ACESSO INT,
  USUARIO_ACESSO VARCHAR(255),
  SENHA VARCHAR(255),
  FOREIGN KEY (ID_PESSOA) REFERENCES PESSOA(ID_PESSOA)
);

-- Criação da tabela VENDA
CREATE TABLE VENDA (
  ID_VENDA INT PRIMARY KEY,
  DATA_VENDA DATE,
  TOTAL_VENDA DECIMAL(10,2),
  STATUS VARCHAR(255),
  NUMERO_CLIENTE INT,
  ENDERECO VARCHAR(255),
  TIPO_LOCAL_ENTREGA VARCHAR(255),
  NUMERO_CASA INT,
  FORMA_PAGAMENTO VARCHAR(255),
  TROCO_PARA DECIMAL(10,2),
  MOTORISTA_ENTREGA INT,
  HORA_SAIDA_MOTORISTA TIME,
  HORA_CHEGADA_MOTORISTA TIME,
  TEMPO_GASTO_TOTAL TIME
);

-- Criação da tabela ITEM_VENDA
CREATE TABLE ITEM_VENDA (
  ID_VENDA INT,
  ITEM_VENDA INT,
  QTD_ITEM INT,
  QUANTIDADE INT,
  VALOR_TL_ITEM DECIMAL(10,2),
  PRIMARY KEY (ID_VENDA, ITEM_VENDA),
  FOREIGN KEY (ID_VENDA) REFERENCES VENDA(ID_VENDA)
);
